IF "accountName" is exist
	PROCESS login
ELSE
	PRINT "silahkan buat akun terlebih dahulu"
	LOAD register page	


IF login process is complete
	LOAD mainPage
	SEARCH "menuName"

	IF "menuName" is exist
		LOAD resultPage
		IF button add to cart clicked
			ADD menu into cart
			IF looking for another menu
				SEARCH another "menuName"
			ELSE
				CALL BuyMenu(cart)
		ELSE
			LOAD buyingPage
			PRINT menu.name
			PRINT menu.price
			CALL BuyMenu(menu)
	ELSE
		print "menu tidak ada"
FINISH
